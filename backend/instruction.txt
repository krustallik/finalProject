Create projects:
npm init -y
npm install koa@^2.13.4
npm install koa-router koa-bodyparser reflect-metadata inversify class-validator class-transformer
npm install --save-dev typescript ts-node @types/node @types/koa @types/koa-router @types/koa-bodyparser
npm run dev

Start projects:
npm install -g typescript
tsc --init
npm i
npx ts-node src/server.ts
npm run dev


DB:
https://www.mongodb.com/try/download/community
npm i mongoose

Hot reload:
npm i ts-node-dev --save-dev
package.json:

"scripts": {
"dev": "ts-node-dev --respawn --transpile-only src/server.ts"
}
